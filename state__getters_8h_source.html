48inline std::vector<double> getState(const std::vector<double>& states, const unsigned char& type_id,
49 const unsigned int i) {
50 sanityCheckStatesSize(states, type_id);
51 unsigned int sample_point_size = getSamplePointSize(states, type_id);
52 unsigned int start = i * getStateDim(type_id);
53 unsigned int end = (i + 1) * getStateDim(type_id);
54 if (i >= sample_point_size || end > getStatesSize(states))
55 throw std::out_of_range("Index i = " + std::to_string(i) + " out of range (" + std::to_string(sample_point_size) +
56 ")");
57
58 return std::vector<double>(states.begin() + start, states.begin() + end);
59}

130inline double approximateTheta(const std::vector<double>& state1, const std::vector<double>& state2,
131 const unsigned char& type_id) {
132 double dx = getX(state2, type_id) - getX(state1, type_id);
133 double dy = getY(state2, type_id) - getY(state1, type_id);
134 return std::atan2(dy, dx);
135}

144inline double getTheta(const std::vector<double>& states, const unsigned char& type_id, const unsigned int i) {
145 sanityCheckStatesSize(states, type_id);
146 if (type_id == DRIVABLE::TYPE_ID || type_id == DRIVABLERWS::TYPE_ID) {
147 std::vector<double> state = getState(states, type_id, i);
148 return getTheta(state, type_id);
149 } else if (type_id == REFERENCE::TYPE_ID) {
150 unsigned int nSamples = getSamplePointSize(states, type_id);
151 std::vector<double> state1, state2;
152 if (i == 0) {
153 state1 = getState(states, type_id, i);
154 state2 = getState(states, type_id, i + 1);
155 } else if (i > 0 && i < nSamples - 1) {
156 state1 = getState(states, type_id, i - 1);
157 state2 = getState(states, type_id, i + 1);
158 } else if (i == nSamples - 1) {
159 state1 = getState(states, type_id, i - 1);
160 state2 = getState(states, type_id, i);
161 }
162 double theta = approximateTheta(state1, state2, type_id);
163 sanityCheckAngle(theta);
164 return theta;
165 } else {
166 throw std::invalid_argument(kExceptionUnknownStateEntry + std::to_string(type_id) + ", " + "theta");
167 }
168}
